## 1) Configuration (generate configs/BreastCA_laplacian2swe_256.yaml from configs/Template-BBDM.yaml)

File paths are specified in the yaml file
Train for 4000 steps, save weights every 1000 steps under results/.../checkpoint/

---------------------------------------------------------------------------------

## 2) Train

conda activate controlnet_new
cd /n/holylfs05/LABS/zhuang_lab/Lab/Jiaming/US2SWEdiff/BBDM
python3 main.py \
  --config configs/BreastCA_laplacian2swe_256.yaml \
  --train \
  --sample_at_start \
  --save_top \
  --gpu_ids 0 \
  --max_steps 4000 \
  --save_step_interval 1000

### Resume training

conda activate controlnet_new
cd /n/holylfs05/LABS/zhuang_lab/Lab/Jiaming/US2SWEdiff/BBDM
CKPT=/n/holylfs05/LABS/zhuang_lab/Lab/Jiaming/US2SWEdiff/BBDM/results/BreastCA_laplacian2swe/BrownianBridge/checkpoint
python3 main.py \
  --config configs/BreastCA_laplacian2swe_256.yaml \
  --train \
  --sample_at_start \
  --save_top \
  --gpu_ids 0 \
  --resume_model "$CKPT/step_model_4000.pth" \
  --resume_optim "$CKPT/step_optim_sche_4000.pth" \
  --max_steps 4000 \
  --save_step_interval 1000
